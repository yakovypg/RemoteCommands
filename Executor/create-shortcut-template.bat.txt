@echo off
set "shortcut=%USERPROFILE%\Desktop\__SHORTCUT-NAME__.lnk"
set "target=explorer.exe"
set "args=__URL__"
set "icon=%~dp0__ICON-NAME__.ico"

powershell -NoProfile -Command ^
 $s=(New-Object -ComObject WScript.Shell).CreateShortcut('%shortcut%'); ^
 $s.TargetPath='%target%'; ^
 $s.Arguments='%args%'; ^
 $s.IconLocation='%icon%'; ^
 $s.Save()
